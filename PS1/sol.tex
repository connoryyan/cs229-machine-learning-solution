    The average empirical loss for logistic regression:
    \[
        J(\theta)=-\frac{1}{m}\sum^m_{i=1}
        {
            y^{(i)}\log{\left(h_\theta (x^{(i)})\right)+
            (1-y^{(i)})\log{\left(1-h_\theta(x^{(i)})\right)}}
        }
    \]
    where $y^{(i)}\in\{0,1\}$, $h_\theta(x)=g(\theta^Tx)$ and $g(z)=1/(1+e^{-z})$. As $g'(z)=g(z)(1-g(z))$, therefore the gradient is
    \[
    \begin{aligned}
        \nabla_\theta J(\theta)
        &=-\frac{1}{m}\sum^m_{i=1}
        {
            y^{(i)}\frac{h_\theta (x^{(i)})(1-h_\theta (x^{(i)}))x^{(i)}}{h_\theta (x^{(i)})}
            +(1-y^{(i)})\frac{-h_\theta (x^{(i)})(1-h_\theta (x^{(i)}))x^{(i)}}{\left(1-h_\theta(x^{(i)})\right)}
        }\\
        &=-\frac{1}{m}\sum^m_{i=1}
        {
            y^{(i)}(1-h_\theta (x^{(i)}))x^{(i)}
            -(1-y^{(i)})h_\theta (x^{(i)})x^{(i)}
        }\\
        &=-\frac{1}{m}\sum^m_{i=1}
        {
            \left(y^{(i)}-h_\theta (x^{(i)})\right)x^{(i)}
        }
    \end{aligned}
    \]
     
     \[
     \begin{aligned}
        H
        &=-\frac{1}{m}\nabla_\theta\sum^m_{i=1}
        {
            \left(y^{(i)}-h_\theta (x^{(i)})\right)x^{(i)}
        }\\
        &=\frac{1}{m}\sum^m_{i=1}
        {
            \nabla_\theta\left(h_\theta (x^{(i)})x^{(i)}\right)
        }\\
        &=\frac{1}{m}\sum^m_{i=1}
        {
            (1-h_\theta (x^{(i)}))(h_\theta (x^{(i)})x^{(i)}(x^{(i)})^T
        }.
    \end{aligned}
    \]

    For any vector $z\in\R^n$, we want to prove
    \[
        z^T Hz\ge0.
    \]
    First take
    \[
    \begin{aligned}
        z^TH
        &=\begin{bmatrix}
            \sum^n_{i=1}z_iH_{i1} & \cdots & \sum^n_{i=1}z_nH_{in}
        \end{bmatrix}
    \end{aligned}
    \]
    Thus, 
    \[
    \begin{aligned}
        z^THz&=\sum^n_{j=1}{z_j\sum^n_{i=1}z_iH_{ij}}\\
        &=\sum^n_{i=1}{\sum^n_{j=1}{z_iz_jH_{ij}}}\\
        &=\frac{1}{m}\sum^m_{k=1}h_\theta(x^{(k)})(1-h_\theta({x^{(k)}}))\sum^n_{i=1}{\sum^n_{j=1}{z_iz_jx^{(k)}_ix^{(k)}_j}}
    \end{aligned}
    \]
    Note that
    \[
        \sum^n_{i=1}\sum^n_{j=1}{z_iz_jx^{(k)}_ix^{(k)}}
        =\left(\sum^n_{i=1}{z_ix^{(k)}_i}\right)\cdot\left(\sum^n_{j=1}{z_jx^{(k)}_j}\right)
        =(z^Tx^{(k)})^2\ge0
    \]
    And note that as $h_\theta\in[0,1]$, $h_\theta(x^{(k)})(1-h_\theta{x^{(k)}})$ is also greater or equal to zero. Hence, $z^THz\ge0$.

        p(y;\lambda)&=\frac{e^{-\lambda}\lambda^y}{y!}\\

\begin{homeworkProblem}
    \textbf{(a)}
    \[
    \begin{aligned}
        \int{p(y;\eta)\dd{y}} &= 1\\
        \frac{\partial}{\partial\eta}\int{p(y;\eta)\dd{y}} &= 0
    \end{aligned}
    \]
    \[
    \begin{aligned}
        \frac{\partial}{\partial\eta}\int{p(y;\eta)\dd{y}}
        &= \int{\frac{\partial}{\partial\eta}p(y;\eta)\dd{y}}\\
        &= \int{b(y)(y-a'(\eta))\exp(\eta y-a(\eta))\dd{y}}\\
        &= \int{yb(y)\exp(\eta y-a(\eta))\dd{y}}
            -a'(\eta)\int{b(y)\exp(\eta y-a(\eta))\dd{y}}\\
        &= \E[y;\eta]-a'(\eta) = 0
    \end{aligned}
    \]

    Hence,
    \[
        \E[Y|X;\theta] = a'(\eta)
    \]

    \textbf{(b)}
    \[
        \begin{aligned}
            \frac{\partial}{\partial\eta}\E[Y|X;\theta]
            &= \frac{\partial}{\partial\eta}\int{yb(y)\exp(\eta y-a(\eta))\dd{y}}\\
            &= \int{\frac{\partial}{\partial\eta}yb(y)\exp(\eta y-a(\eta))\dd{y}}\\
            &= \int yb(y)(y-a'(\eta))\exp(\eta y-a(\eta))\dd{y}\\
            &= \int{y^2b(y)\exp(\eta y-a(\eta))\dd{y}}
            -a'(\eta)\int y{b(y)\exp(\eta y-a(\eta))\dd{y}}\\
            &= \E[Y^2|X;\theta] + \E[Y|X;\theta]^2\\
            &= \Var[Y^2|X;\theta]
        \end{aligned}
    \]
    And note that $\frac{\partial}{\partial\eta}\E[Y|X;\theta] = a''(\eta)$, therefore,
    \[
        \Var[Y|X;\theta] = a''(\eta)
    \]

    \textbf{(c)}
    Take one example $(x^{(k)},y^{(k)})$ from the data set, the gradient of the loss function is
    \[
    \begin{aligned}
        \nabla_\theta\ell(\theta) &= -\nabla_\theta \log\left( b(y^{(k)})\exp(\theta^Tx^{(k)} y^{(k)}-a(\theta^Tx^{(k)})) \right)\\
        &= -\nabla_\theta \left( \log[b(y^{(k)})] + \theta^Tx^{(k)} y^{(k)}-a(\theta^Tx^{(k)}) \right)\\
        &= -x^{(k)}y^{(k)}+x^{(k)}a'(\eta)
    \end{aligned}
    \]
    Thus, the Hessian is
    \[
    \begin{aligned}
        H &= \nabla_\theta [-x^{(k)}y^{(k)}+x^{(k)}a'(\eta)]\\
        &= a''(\eta)x^{(k)}(x^{(k)})^T
    \end{aligned}
    \]

    For any vector $z$ that has the same dimension as $x$,
    \[
    \begin{aligned}
            z^THz &= \sum_i\sum_jz_iH_{ij}z_j\\
            &= a''(\eta)\sum_i\sum_jz_ix^{(k)}_ix^{(k)}_jz_j\\
            &= a''(\eta)\sum_iz_ix^{(k)}_i\sum_jx^{(k)}_jz_j\\
            &= a''(\eta)(z^Tx^{(k)})^2\\
            &= (z^Tx^{(k)})^2\Var[Y|X;\theta] \ge 0.
    \end{aligned}
    \]
    Note that $\ell(\theta)=\sum_k\log (-b(y^{(k)})\exp(\theta^Tx^{(k)} y^{(k)}-a(\theta^Tx^{(k)}))$. As each term has a positive Hessian, the sum also has a positive Hessian. That is to say, the loss function $\ell(\theta)$ is PSD.

        &= \sum_{i = 1} ^ m {
            \sum_{j = 1} ^ m{
                (\theta^T x^{(i)} - y^{(i)}) W_{ii} (\theta^T x^{(j)} - y^{(j)})
            }
        }\\
        &= (X\theta-y)^TW(X\theta-y),
    \end{aligned}
    \]
    where $X\in \R^{m\times n}$ has entries $X_{ij}=x^{(i)}_j$.

    \textbf{ii.}
    \[
    \begin{aligned}
        \nabla_\theta J(\theta)
        &= \nabla_\theta(X\theta-y)\nabla_{X\theta-y}[(X\theta-y)^TW(X\theta-y)]\\
        &= X^T(\nabla_{X\theta-y}(X\theta-y)W(X\theta-y)

        &= X^T[W(X\theta-y) + W^T(X\theta-y)]\\
        &= X^T(W+W^T)(X\theta-y)\\
        &= 2X^TW(X\theta-y), \text{(as W is a diag matrix)}
    \end{aligned}
    \]
    Let
    \[
    \begin{aligned}
        \nabla_\theta J(\theta) &= 0\\
        X^TWX\theta &= X^TWy\\
        \theta &= (X^TWX)^{-1}X^TWy
    \end{aligned}
    \]
